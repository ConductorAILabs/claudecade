║  CLAUDECADE ENGINE  ·  claudcade_engine.py  ·  zero dependencies             ║
╚══════════════════════════════════════════════════════════════════════════════╝

  A terminal game engine built on Python's curses library.
  Copy claudcade_engine.py into your project and import it.
  Tag @ConductorAILabs on social when you ship.

  INSTALL
  ───────
  No pip install. Just copy the file:

    cp claudcade_engine.py  your-project/

  QUICK START
  ───────────
  from claudcade_engine import Engine, Scene, Input, Renderer
  from claudcade_engine import PLAYER, ENEMY, GOOD, GOLD, NEUTRAL

  class Game(Scene):
      def on_enter(self):
          self.score = 0

      def update(self, inp: Input, tick: int, dt: float) -> str | None:
          if inp.pause:           return "quit"
          if inp.just_pressed(ord('j')):
              self.score += 1

      def draw(self, r: Renderer, tick: int):
          r.header("MY GAME", left=f"SCORE {self.score}")
          r.center(r.H // 2, "Hello terminal!", PLAYER, bold=True)

  Engine("My Game", fps=30).scene("game", Game()).run("game")

╔══════════════════════════════════════════╗
║  COLOR CONSTANTS                         ║
╚══════════════════════════════════════════╝

  Use semantic names — they tell you what the color is FOR:

    PLAYER   = CYAN       # player character, info panels
    ENEMY    = RED        # enemies, damage, warnings
    GOOD     = GREEN      # health, positives, pickups
    GOLD     = YELLOW     # score, coins, fire, towns
    NEUTRAL  = WHITE      # borders, neutral text
    SPECIAL  = MAGENTA    # magic, abilities
    SELECT   = HIGHLIGHT  # selected menu item (black on white)
    WATER    = BLUE       # water, mana

  Raw names (same values, avoid where possible):
    CYAN  RED  GREEN  YELLOW  WHITE  MAGENTA  HIGHLIGHT  BLUE

  Pass to any Renderer method:
    r.text(row, col, "HIT!", ENEMY, bold=True)
    r.bar(row, col, hp, max_hp, 20, fill_color=GOOD)

╔══════════════════════════════════════════╗
║  INPUT                                   ║
╚══════════════════════════════════════════╝

  The inp object is passed to Scene.update() every frame.

  MOVEMENT ALIASES
  ────────────────
    inp.up      inp.down     inp.left     inp.right
    WASD or arrow keys — True every frame the key is held.

  ACTION ALIASES
  ──────────────
    inp.fire      J, F, or mouse click
    inp.jump      SPACE
    inp.confirm   ENTER or SPACE
    inp.pause     ESC

  SINGLE-FRAME DETECTION  ← use this for menus and one-shot actions
  ──────────────────────────────────────────────────────────────────
    inp.just_pressed(ord('j'))    True only the FIRST frame key goes down
    inp.just_released(ord('j'))   True only the FIRST frame key comes up

    # Wrong — fires every frame the key is held (cursor races through menu):
    if inp.down: cursor += 1

    # Right — fires once per keypress:
    if inp.just_pressed(curses.KEY_DOWN, ord('s')): cursor += 1

  RAW CHECK
  ─────────
    inp.pressed(ord('a'), ord('A'))    True any frame any of these are held

  MOUSE
  ─────
    inp.mouse_click    True if left mouse button was clicked this frame

╔══════════════════════════════════════════╗
║  RENDERER                                ║
╚══════════════════════════════════════════╝

  r is passed to Scene.draw() every frame.
  Coordinates are (row, col). Row 0 = TOP of terminal.
  All methods are bounds-safe — drawing outside the terminal does nothing.

  DIMENSIONS
  ──────────
    r.H    terminal height in rows
    r.W    terminal width in columns

  PRIMITIVES
  ──────────
    r.text(row, col, "string", color, bold, dim, reverse, blink)
    r.center(row, "string", color, bold)
    r.fill(row, col, h, w, char=' ', color)   # clear a region

  SHAPES
  ──────
    r.box(row, col, h, w, color, title="", double=True)
      ╔═══════╗   double=True  (default)
      ┌───────┐   double=False

    r.outer_border(color)   full terminal border

  PROGRESS BARS
  ─────────────
    r.bar(row, col, value, maximum, width, fill_color, empty_color, label)
      ████████░░░░  HP 48/60

    r.vbar(top_row, col, height, value, maximum, fill_color, empty_color)
      Vertical bar — top=empty, bottom=full. Good for power meters.

  SPRITES AND ENTITIES
  ────────────────────
    r.sprite(row, col, lines, color, bold)
      lines = ['( o )', '|=|', '/ \\']

    r.at(entity, sprite, color, bold)
      Draws sprite at entity.x, entity.y — no manual (row,col) swap needed.

    r.menu(row, col, options, cursor, color_selected, color_normal, width)
      Vertical menu with ▸ cursor. Cursor wraps automatically.

  TEXT
  ────
    r.wrapped_text(row, col, text, width, color, bold)
      Word-wraps text into width columns. Returns number of rows used.

    r.dialog(row, col, width, speaker, text, tick, speaker_color, portrait, prompt)
      RPG dialog box: name tag, optional portrait, wrapped text, blinking ▼.

  STANDARD SCREENS
  ────────────────
    r.header(game_name, left="", right="", color)
      ╔══════════════════════════════════╗
      ║▓▒░ ★ GAME NAME ★  LEFT    RIGHT ░▒▓║
      ╠══════════════════════════════════╣
      Draws rows 0–2. Call first in draw().

    r.footer(controls, color)
      ╠══════════════════════════════════╣
      ║ WASD Move  J Fire  ESC Pause    ║
      Draws rows H-3 and H-2.

    r.pause_overlay(game_name, controls)
      Floating pause menu. Draw after game scene so game shows behind.

    r.gameover_screen(title, score_line, player_label, rank, tick, prompt, title_color)
      Full-screen game over or victory screen with blinking prompt.
      title_color=ENEMY for game over, GOOD for victory.

    r.stars(star_list)
      Draw parallax star field. Use with make_stars() and scroll_stars().

╔══════════════════════════════════════════╗
║  SCENE                                   ║
╚══════════════════════════════════════════╝

  One screen of a game. Subclass and implement three methods.

  class GameScene(Scene):

      def on_enter(self):
          # Called once when scene becomes active.
          # self.engine.H and self.engine.W are available here.
          self.player = Player(self.engine.W // 2, self.engine.H // 2)

      def update(self, inp: Input, tick: int, dt: float) -> str | None:
          # Called every frame. dt = elapsed seconds (≈0.033 at 30fps).
          # Return a scene name to switch, "quit" to exit, or None.
          self.player.x += 10 * dt   # always scale by dt for smooth movement
          if inp.pause:
              return "pause"
          if self.player.dead:
              return "gameover"

      def draw(self, r: Renderer, tick: int):
          # Called every frame after update(). Draw background first.
          r.header("MY GAME")
          r.at(self.player, self.player.sprite())

  SWITCHING SCENES
  ────────────────
    return "gameover"               # from update() — immediate switch
    self.engine.switch("gameover")  # programmatic — same result
    self.engine.fade_to("gameover", ticks=12)  # with fade transition

  DRAW ORDER (critical for correct rendering)
  ────────────────────────────────────────────
    1. background / stars / terrain
    2. world entities (enemies, bullets, platforms)
    3. player
    4. particles
    5. HUD (header, bars, score)
    6. overlays (pause, dialog)

╔══════════════════════════════════════════╗
║  ENGINE                                  ║
╚══════════════════════════════════════════╝

  Engine("Title", fps=30)
    .scene("menu",     MenuScene())
    .scene("game",     GameScene())
    .scene("gameover", GameOverScene())
    .run("menu")

  PROPERTIES (read from scenes via self.engine.*)
  ─────────────────────────────────────────────────
    engine.H        terminal height (updated every frame)
    engine.W        terminal width
    engine.dt       seconds since last frame (≈0.033 at 30fps)

  METHODS
  ───────
    engine.switch(name)            immediate scene switch
    engine.fade_to(name, ticks)    scene switch with fade transition

  BUILT-IN BEHAVIOR
  ─────────────────
    · Terminal size is checked every frame. If too small (< 50×18)
      a resize message is shown instead of running the game.
    · Unknown scene names print a warning to stderr (no silent failures).
    · Frame timing never tries to "catch up" — if a frame runs long,
      the next target time is adjusted, preventing speed bursts.
    · Returns "quit" from update() to cleanly exit.

╔══════════════════════════════════════════╗
║  ENTITY  /  PHYSICS ENTITY               ║
╚══════════════════════════════════════════╝

  Entity(x, y)
    Base class. x=col, y=row in screen space.

    Properties: x, y, vx, vy, alive

    entity.rect()             (x, y, w, h) — override for custom hitbox
    entity.collides(other)    AABB collision check

  PhysicsEntity(x, y)
    Adds gravity, velocity, and floor collision.

    entity.gravity = 1.5      class or instance attribute
    entity.grounded           True when resting on the floor

    entity.apply_physics(floor_y, dt)
      Call every frame. Always pass dt:
        self.apply_physics(floor_y=H - 4, dt=dt)

  DRAW CONVENTION
  ───────────────
    r.at(entity, sprite)   # recommended — handles row/col conversion
    r.sprite(int(entity.y), int(entity.x), lines)   # manual

  ENTITY LISTS
  ────────────
    # Update all, then cull dead ones:
    for e in self.enemies:
        e.update(inp, tick, dt)
    self.enemies = [e for e in self.enemies if e.alive]

    # Check player vs all enemies:
    for e in self.enemies:
        if self.player.collides(e):
            self.player.alive = False

╔══════════════════════════════════════════╗
║  ANIMSPRITE                              ║
╚══════════════════════════════════════════╝

  Frame-cycling sprite with named states and optional completion callbacks.

  SETUP
  ─────
    states = {
        'idle': [ ['( o )', '|=|', '/ \\'] ],          # 1 frame
        'run':  [ ['(_o_)', '/|\\', '/ \\'],            # 2 frames
                  ['(_o_)', '\\|/', '/ \\'] ],
        'die':  [ ['( x )', ' | ', '/ \\'],
                  ['( x )', '   ', ' - '] ],
    }
    sprite = AnimSprite(states, ticks_per_frame=6)

  EACH FRAME
  ──────────
    sprite.set_state('run')         # switch state (resets to frame 0)
    sprite.tick()                   # advance animation
    r.sprite(row, col, sprite.current(), PLAYER)

  ONE-SHOT ANIMATIONS  (fire-and-forget with callback)
  ─────────────────────────────────────────────────────
    sprite.set_state('die', on_complete=lambda: self.remove())
    sprite.set_state('attack', force=True)   # restart even if already in state

╔══════════════════════════════════════════╗
║  CAMERA                                  ║
╚══════════════════════════════════════════╝

  Smooth-follow camera for scrolling worlds.

  SETUP
  ─────
    self.cam = Camera()
    self.cam.set_bounds(0, world_width)   # prevent scrolling past edges

  EACH FRAME  (in update)
  ────────────────────────
    self.cam.follow(player.x, player.y, lerp=0.12)
    # lerp: 1.0 = instant snap, 0.05 = very smooth

  DRAW ENTITIES  (in draw)
  ─────────────────────────
    row, col = self.cam.world_to_screen(entity.x, entity.y, r.W, r.H)
    r.sprite(row, col, entity.sprite())

  Y-AXIS CONVENTION
  ─────────────────
    y_up=False  (default)  Y increases downward — terminal convention
                           Use for top-down games, maps, shooters
    y_up=True              Y increases upward — platformer convention
                           Y=0 is the floor, higher Y = higher up

    row, col = cam.world_to_screen(wx, wy, W, H, y_up=True)

  VISIBILITY CHECK
  ─────────────────
    if cam.on_screen(entity.x, entity.y, W, H):
        # only draw visible entities (performance)

╔══════════════════════════════════════════╗
║  PARTICLES                               ║
╚══════════════════════════════════════════╝

  Moving particle system for explosions, sparks, and effects.

  SETUP
  ─────
    self.particles = Particles()

  EACH FRAME
  ──────────
    self.particles.update()     # advance all particles
    self.particles.draw(r)      # render (call after entities, before HUD)

  SPAWNING
  ────────
    # Radial explosion burst:
    particles.explode(col, row, count=8, color=GOLD, chars='✦✧★·◆')

    # Directional burst (exhaust, hit sparks):
    particles.burst(col, row, direction=(-1, 0), spread=0.5, count=4, color=ENEMY)

    # Single custom particle:
    particles.spawn(col, row, vx=-0.5, vy=-1.2, char='★', color=GOLD, life=10)

    # Count active:
    particles.count

╔══════════════════════════════════════════╗
║  TIMER                                   ║
╚══════════════════════════════════════════╝

  Countdown timer for delays, cooldowns, and intervals.

  USAGE
  ─────
    # One-shot: fires once after 2 seconds
    self.invincible = Timer(2.0)

    # Auto-repeating: fires every 0.5 seconds
    self.fire_rate = Timer(0.5, auto_reset=True)

  EACH FRAME  (in update)
  ────────────────────────
    if self.invincible.tick(dt):
        self.can_take_damage = True

    if self.fire_rate.tick(dt):
        self.bullets.append(Bullet(x, y))

  PROPERTIES
  ──────────
    timer.done          True if timer has completed
    timer.frac          Progress 0.0 → 1.0
    timer.remaining     Seconds left

  RESET
  ─────
    timer.reset()               restart with same duration
    timer.reset(duration=1.5)   restart with new duration

  MINIMAL COMPLETE GAME
  ──────────────────────
    from claudcade_engine import *

    class Menu(Scene):
        def update(self, inp, tick, dt):
            if inp.confirm: return "game"
            if inp.pause:   return "quit"
        def draw(self, r, tick):
            r.center(r.H//2, "PRESS SPACE", GOLD, bold=True)

    class Game(Scene):
        def on_enter(self):
            self.x = self.engine.W // 2
            self.score = 0
        def update(self, inp, tick, dt):
            if inp.right: self.x += 20 * dt
            if inp.left:  self.x -= 20 * dt
            if inp.pause: return "menu"
        def draw(self, r, tick):
            r.header("GAME", right=f"SCORE {self.score}")
            r.text(r.H//2, int(self.x), "(◙)", PLAYER, bold=True)

    Engine("Demo").scene("menu", Menu()).scene("game", Game()).run("menu")

  ENTITY LIST WITH COLLISIONS
  ────────────────────────────
    class GameScene(Scene):
        def on_enter(self):
            self.player  = Player(10, 10)
            self.enemies = [Enemy(x, 5) for x in range(20, 60, 8)]
            self.bullets : list[Bullet] = []
            self.particles = Particles()

        def update(self, inp, tick, dt):
            self.player.update(inp, tick, dt)

            if inp.just_pressed(ord('j')):
                self.bullets.append(Bullet(self.player.x, self.player.y))

            for b in self.bullets: b.update(inp, tick, dt)
            for e in self.enemies: e.update(inp, tick, dt)
            self.particles.update()

            # bullet vs enemy collisions
            for b in self.bullets:
                for e in self.enemies:
                    if b.collides(e):
                        b.alive = False
                        e.alive = False
                        self.particles.explode(int(e.x), int(e.y), color=ENEMY)

            # cull dead
            self.bullets = [b for b in self.bullets if b.alive]
            self.enemies = [e for e in self.enemies if e.alive]

            if not self.player.alive: return "gameover"

        def draw(self, r, tick):
            r.header("SHOOTER")
            for e in self.enemies: r.at(e, '(×)', ENEMY)
            for b in self.bullets: r.at(b, '─▶', GOLD)
            r.at(self.player, '(◙)', PLAYER, bold=True)
            self.particles.draw(r)

  SCROLLING WORLD WITH CAMERA

    class World(Scene):
        def on_enter(self):
            self.cam    = Camera()
            self.cam.set_bounds(0, 500)
            self.player = Player(0, self.engine.H // 2)
            self.world_width = 500

        def update(self, inp, tick, dt):
            self.player.update(inp, tick, dt)
            self.cam.follow(self.player.x, self.player.y)

        def draw(self, r, tick):
            r.header("WORLD")
            for entity in self.entities:
                if self.cam.on_screen(entity.x, entity.y, r.W, r.H):
                    row, col = self.cam.world_to_screen(entity.x, entity.y, r.W, r.H)
                    r.sprite(row, col, entity.sprite(), entity.color)

    # Wrong — speed changes with FPS:
    entity.x += 2

    entity.x += 2 * dt * 30    # "2 cols per frame at 30fps reference"

    # Physics:
    entity.apply_physics(floor_y=H - 3, dt=dt)
